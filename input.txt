import io;

class Arithmetic_less {
	public static void main(String[] args) {
        int x;
        boolean big;
        x=1;
        big = false;
        while(x<10){
            if(!(x<5) && !big){
                big = true;
            }
            else{
                big = false;
            }
            io.println(big);
            x = x + 1;
        }
	}
}
